exports.checkLogin = function(req, res, next, loggedInUsers, isTesting) {
  if (req.user && loggedInUsers[req.user.username]) {
    if (isTesting) {
      res.json({"statusCode": 401, "message": "Test is running."});
    }else {
      next();
    }
  }else {
    res.status(401).end("Unauthorized");
  }
}
